2. Prime Number :


        class PrimeNo {
            public static void main(String[] args) {
                int no = 20;
                int status=0;
                for(int i=2; i<=no/2;i++){
                    if(no%i==0){
                        status=1;
                        break;
                    }
                }
                    if(status==0){
                        System.out.println("No. is prime");
                    }
                    else{
                        System.out.println("No. is not prime");
                    }
            }
        }

4. Palindrome Number :

    
        public class Palindrome_Num {

            public static void main(String[] args) {
                int n = 34443, rem;
                int pal = 0;
                int m = n;
                while (n > 0) {
                    rem = n % 10;
                    n = n / 10;
                    pal = pal * 10 + rem;
                }
                if (m == pal) {
                    System.out.println("Number is palindrome:" + pal);
                } else {
                    System.out.println("Number is not palindrome:" + pal);
                }

            }

        }

12. Half * pyramid pattern in java


        class Pyramid {
            public static void main(String[] args) {
                int rows = 5;
                for (int i = 1; i <= rows; i++) {
                    for (int j = 1; j <= i; j++) {
                        System.out.print("*");
                    }
                    System.out.println("");
                }
            }
        }
